SELECT 
    s.name AS 'Wr',
    ss.score AS 'ΘZ',
    (SELECT COUNT(DISTINCT score) FROM student_scores WHERE score > ss.score) + 1 AS 'WΩ'
FROM 
    students s
JOIN 
    class_student cs ON s.school_num = cs.school_num
JOIN 
    student_scores ss ON s.school_num = ss.school_num
WHERE 
    cs.class_code = '102'
ORDER BY 
    ss.score DESC;
